is_county_safe = {
    # it's safe if
    OR = {
        # I control it
        this = root

        # or my loyal vassal controls it
        AND = {
            is_vassal_or_below_of = root
            NOT = { is_at_war_with = root }
        }

        # or someone who
        NOR = {
            # is not at war with me
            is_at_war_with = root
            # their liege isn't at war with me
            any_liege_or_above = {
                is_at_war_with = root
            }
        }
    }
}
