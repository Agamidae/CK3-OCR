add_neighbours_for_vassal = {
	holder = {
		if = {
			limit = {
				NOR = {
					is_in_list = adjacent_rulers
					is_in_list = adjacent_rulers_across_water
					is_in_list = adjacent_vassals
				}
			}
			if = {
				limit = {
					is_independent_ruler = no
					liege = root.liege
				}
				add_to_list = adjacent_vassals
			}
			if = {
				limit = {
					is_independent_ruler = no
					top_liege = root.top_liege
				}
				every_liege_or_above = {
					limit = {
						is_independent_ruler = no
						liege = root.liege
					}
					add_to_list = adjacent_vassals
				}
			}
			if = {
				limit = {
					is_independent_ruler = no
					NOT = {
						top_liege = root.top_liege
					}
				}
				top_liege = {
					add_to_list = $x$
				}
			}
			if = {
				limit = {
					is_independent_ruler = yes
					NOT = { this = root.top_liege }
				}
				add_to_list = $x$
			}
			set_variable = {
				name = direction
				value = capital_province.ruler_direction_svalue
			}
			if = {
				limit = {
					var:direction = 0
				}
				set_variable = {
					name = direction
					value = capital_province.direction_svalue
				}
			}
		}
	}
}
