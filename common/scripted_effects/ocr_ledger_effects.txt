add_to_ledger = {
	ordered_in_list = {
		list = $list$
		order_by = $x$
		min = 0

		root = {
			add_to_variable_list = {
				name = $list$
				target = prev
			}
		}
	}
}

add_to_ledger_reverse = {
	ordered_in_list = {
		list = $list$
		order_by = {
			value = $x$
			multiply = -1
		}
		min = 0

		root = {
			add_to_variable_list = {
				name = $list$
				target = prev
			}
		}
	}
}
