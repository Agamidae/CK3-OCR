on_travel_plan_movement = {
    on_actions = { ocr_travel }
}

ocr_travel = {
    effect = {
        if = {
            limit = {
                location = {
                    is_a_point_of_interest = yes
                }
            }
            add_to_variable_list = {
                name = visited_points_of_interest
                target = location
            }
        }
    }
}
