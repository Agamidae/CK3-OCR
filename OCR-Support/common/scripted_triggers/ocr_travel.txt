has_filter = {
    root = {
        is_target_in_variable_list = {
            name = $LIST$
            target = flag:$VAR$
        }
    }
}

has_filter_tour = {
    has_filter = { LIST = filters_tour VAR = $VAR$ }
}

has_filter_tour_scope = {
    root = {
        is_target_in_variable_list = {
            name = filters_tour
            target = scope:target
        }
    }
}
