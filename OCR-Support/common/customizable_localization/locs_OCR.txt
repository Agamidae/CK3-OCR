army_grouping = {
	type = character
	text = {
		trigger = {
			OR = {
				NOT = {
					exists = var:army_grouping
				}
				var:army_grouping = 0
			}
		}
		localization_key = holding
	}
	text = {
		trigger = {
			exists = var:army_grouping
			var:army_grouping = 1
		}
		localization_key = county
	}
	text = {
		trigger = {
			exists = var:army_grouping
			var:army_grouping = 2
		}
		localization_key = duchy
	}
	text = {
		trigger = {
			exists = var:army_grouping
			var:army_grouping = 3
		}
		localization_key = kingdom
	}
	text = {
		trigger = {
			exists = var:army_grouping
			var:army_grouping = 4
		}
		localization_key = empire
	}
}
