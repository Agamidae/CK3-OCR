character_health = {
	type = character
	text = {
		trigger = {
			health < dying_health
		}
		localization_key = HEALTH_DYING
	}
	text = {
		trigger = {
			health >= dying_health
			health < poor_health
		}
		localization_key = HEALTH_NEAR_DEATH
	}
	text = {
		trigger = {
			health >= poor_health
			health < fine_health
		}
		localization_key = HEALTH_POOR
	}
	text = {
		trigger = {
			health >= fine_health
			health < excellent_health
		}
		localization_key = HEALTH_FINE
	}
	text = {
		trigger = {
			health > excellent_health
		}
		localization_key = HEALTH_EXCELLENT
	}
}
