num_faith_rulers = {
  every_ruler = {
    limit = { faith = root }
    add = 1
  }
}

num_faith_followers = {
  every_living_character = {
    limit = { faith = root }
    add = 1
  }
}

num_faith_sinners = {
  every_ruler = {
    limit = {
      faith = root
      any_character_trait = {
        root = {
          trait_is_sin = prev
        }
      }
    }
    add = 1
  }
}

num_faith_holy_war_attacker = {
  every_ruler = {
    limit = {
      faith = root
      is_at_war = yes
      any_character_war = {
        is_war_leader = prev
        is_attacker = prev
        OR = {
          using_cb = minor_religious_war
          using_cb = religious_war
          using_cb = major_religious_war
          using_cb = flowery_war_cb
        }
      }
    }
    add = 1
  }
}
