add_to_ledger = {
	ordered_in_list = {
		list = $list$
		order_by = $x$
		min = 0

		root = {
			add_to_variable_list = {
				name = $list$
				target = prev
			}
		}
	}
}

add_to_ledger_reverse = {
	ordered_in_list = {
		list = $list$
		order_by = {
			value = $x$
			multiply = -1
		}
		min = 0

		root = {
			add_to_variable_list = {
				name = $list$
				target = prev
			}
		}
	}
}

#### rulers

add_landed_rulers = {
	clear_global_variable_list = $x$
	ordered_independent_ruler = {
		limit = {
			sub_realm_size > 0
			exists = capital_province
			capital_province = {
				geographical_region = $x$
			}
		}
		order_by = sub_realm_size
		min = 0

		add_to_global_variable_list = {
			name = $x$
			target = this
		}
	}
}


add_playable_rulers = {
	clear_global_variable_list = $x$
	ordered_independent_ruler = {
		limit = {
			sub_realm_size > 0
			exists = capital_province
			NOR = {
				has_government = theocracy_government
				has_government = republic_government
				has_government = mercenary_government
				has_government = holy_order_government
			}
			capital_province = {
				geographical_region = $x$
			}
		}
		order_by = sub_realm_size
		min = 0

		add_to_global_variable_list = {
			name = $x$
			target = this
		}
	}
}

count_playable_rulers = {
	every_ruler = {
		limit = {
			exists = capital_province
			NOR = {
				has_government = theocracy_government
				has_government = republic_government
				has_government = mercenary_government
				has_government = holy_order_government
			}
			capital_province = {
				geographical_region = $x$
			}
		}
		add = 1
	}
}
