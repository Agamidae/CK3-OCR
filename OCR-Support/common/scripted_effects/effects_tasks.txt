council_task_effect = {
	scope:councillor = {
    	set_council_task = {
    		task_type = $task$
    		target = scope:county
    	}
    	hidden_effect = {
    		set_council_task = {
    			task_type = $task$
    			target = scope:county.title_province
    		}
    	}
    }
}

court_task_effect = {
	scope:councillor = {
		# this shows up in the tooltip
    	set_council_task = {
    		task_type = $task$
    		target = scope:target_character
    	}
		# this actually starts the task but causes errors
    	hidden_effect = {
    		set_council_task = {
    			task_type = $task$
    			target = scope:target_character
    		}
    	}
    }
}
