refresh_region_province = {
	effect = {
		scope:player = {
			set_variable = {
				name = region_province
				value = scope:target
			}
		}
		scope:target = {
			create_region_lists = yes
		}
	}
}

clear_region_province = {
	effect = {
		remove_variable = region_province
	}
}
