faith_map_extents = {
  effect = {
    clear_variable_list = counties

    every_county = {
      limit = {
        is_landless_type_title = no
        faith = root
      }
      root = {
        add_to_variable_list = {
          name = counties
          target = prev
        }
      }
    }

    map_extents_faith_ocr = {
      list = counties
      faith = faith
    }
  }
}
