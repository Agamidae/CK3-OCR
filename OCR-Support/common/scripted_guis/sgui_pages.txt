next_page = {
	effect = {
        switch = {
            trigger = scope:page
            flag:rules_page = {
                next_page_effect = {
                    VAR = rules_page
                    ITEMS = 20
                }
            }
        }
	}
}

prev_page = {
    effect = {
        switch = {
            trigger = scope:page
            flag:rules_page = {
                prev_page_effect = {
                    VAR = rules_page
                    ITEMS = 20
                }
            }
        }
	}
	is_valid = {
        OR = {
            AND = {
                scope:page = flag:rules_page
                exists = var:rules_page
                var:rules_page > 19
            }
        }
	}
}

reset_page = {
	effect = {
        switch = {
            trigger = scope:page
            flag:rules_page = {
                set_variable = {
                    name = rules_page
                    value = 0
                }
            }
        }
	}
}

test_page = {

            name = test_page
            value = 10
        }
    }
}

		set_variable = {

			}

		if = {
			limit = {
				var:marriage_page_size = 10
			}
			set_variable = {
				name = marriage_page_size
				value = 5
			}
		}
		else = {
			set_variable = {
				name = marriage_page_size
				value = 10
			}
			if = {
				limit = {
					var:marriage_page > 4
					marriage_page_modulo > 0
				}
				change_variable = {
					name = marriage_page
					subtract = 5
				}
			}
		}
	}
}
