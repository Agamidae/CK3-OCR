next_page = {
	effect = {
        switch = {
            trigger = scope:page
            flag:rules_page = {
                next_page_effect = {
                    VAR = rules_page
                    ITEMS = 20
                }
            }
        }
	}
}

prev_page = {
    effect = {
        switch = {
            trigger = scope:page
            flag:rules_page = {
                prev_page_effect = {
                    VAR = rules_page
                    ITEMS = 20
                }
            }
        }
	}
	is_valid = {
        OR = {
            AND = {
                scope:page = flag:rules_page
                exists = var:rules_page
                var:rules_page > 19
            }
        }
	}
}

reset_page = {
	effect = {
        switch = {
            trigger = scope:page
            flag:rules_page = {
                set_variable = {
                    name = rules_page
                    value = 0
                }
            }
        }
	}
}

test_page = {
    effect = {
        set_global_variable = {
            name = test_page
            value = 10
        }
    }
}


### admin

next_admin_page = {
	effect = {
		if = {
			limit = {
				NOT = {
					exists = var:admin_page
				}
			}
			set_variable = {
				name = admin_page
				value = 0
			}
		}
		change_variable = {
			name = admin_page
			add = 10
		}
	}
}

prev_admin_page = {
	effect = {
		if = {
			limit = {
				NOT = {
					exists = var:admin_page
				}
			}
			set_variable = {
				name = admin_page
				value = 0
			}
		}
		change_variable = {
			name = admin_page
			add = -10
		}
	}
	is_valid = {
		exists = var:admin_page
		var:admin_page > 9
	}
}

reset_admin_page = {
	effect = {
		set_variable = {
			name = admin_page
			value = 0
		}
	}
}

				}
			}

			}
		}
		change_variable = {
			name = marriage_page
			add = var:marriage_page_size
		}
	}
}

prev_marriage_page = {
	effect = {
		if = {
			limit = {
				NOT = {
					exists = var:marriage_page
				}
			}
			set_variable = {
				name = marriage_page
				value = 0
			}
		}
		change_variable = {
			name = marriage_page
			subtract = var:marriage_page_size
		}
	}
	is_valid = {
		exists = var:marriage_page
		var:marriage_page >= var:marriage_page_size
	}
}

reset_marriage_page = {
	effect = {
		set_variable = {
			name = marriage_page
			value = 0
		}
		if = {
			limit = {
				NOT = {
					has_variable = marriage_page_size
				}
			}
			set_variable = {
				name = marriage_page_size
				value = 10
			}
		}
	}
}

resize_marriage_page = {
	effect = {
		if = {
			limit = {
				var:marriage_page_size = 10
			}
			set_variable = {
				name = marriage_page_size
				value = 5
			}
		}
		else = {
			set_variable = {
				name = marriage_page_size
				value = 10
			}
			if = {
				limit = {
					var:marriage_page > 4
					marriage_page_modulo > 0
				}
				change_variable = {
					name = marriage_page
					subtract = 5
				}
			}
		}
	}
}
