adjacent_counties = {
	# root is county
	saved_scopes = {
		target # province
		player
	}
	effect = {
		if = {
			limit = {
				# NOT = { has_variable_list = adjacent_counties }
				# is_landed = yes
				always = yes
			}
			scope:target = {
				save_scope_as = origin
			}
			clear_variable_list = adjacent_counties
			clear_variable_list = adjacent_counties_across_water
			clear_variable_list = adjacent_seas
			clear_variable_list = adjacent_rivers

			# every_county_province
			# scope:target
			every_county_province = {
				ordered_neighboring_province = {
					limit = {
						is_sea_province = yes
						root = {
							NOT = {
								is_target_in_variable_list = {
									name = adjacent_seas
									target = prev
								}
							}
						}
					}
					order_by = direction_svalue
					min = 0
					root = {
						add_to_variable_list = {
							name = adjacent_seas
							target = prev
						}
					}
				}
				ordered_neighboring_province = {
					limit = {
						is_river_province = yes
						root = {
							NOT = {
								is_target_in_variable_list = {
									name = adjacent_rivers
									target = prev
								}
							}
						}
					}
					order_by = direction_svalue
					min = 0
					root = {
						add_to_variable_list = {
							name = adjacent_rivers
							target = prev
						}
					}
				}
			}

			# maybe move this to another sgui
			scope:target = {
				clear_variable_list = adjacent_seas
				clear_variable_list = adjacent_rivers
				clear_variable_list = adjacent_provinces
				ordered_neighboring_province = {
					limit = {
						is_sea_province = yes
						scope:target = {
							NOT = {
								is_target_in_variable_list = {
									name = adjacent_seas
									target = prev
								}
							}
						}
					}
					order_by = direction_svalue
					min = 0
					scope:target = {
						add_to_variable_list = {
							name = adjacent_seas
							target = prev
						}
					}
				}
				ordered_neighboring_province = {
					limit = {
						is_river_province = yes
						scope:target = {
							NOT = {
								is_target_in_variable_list = {
									name = adjacent_rivers
									target = prev
								}
							}
						}
					}
					order_by = direction_svalue
					min = 0
					scope:target = {
						add_to_variable_list = {
							name = adjacent_rivers
							target = prev
						}
					}
				}
				ordered_neighboring_province = {
					limit = {
						has_holding = yes
						scope:target = {
							NOT = {
								is_target_in_variable_list = {
									name = adjacent_provinces
									target = prev
								}
							}
						}
					}
					order_by = direction_svalue
					min = 0
					scope:target = {
						add_to_variable_list = {
							name = adjacent_provinces
							target = prev
						}
					}
				}
			}

			ordered_neighboring_county = {
				order_by = title_province.direction_svalue
				min = 0
				root = {
					add_to_variable_list = {
						name = adjacent_counties
						target = prev
					}
				}
			}
			ordered_title_to_title_neighboring_and_across_water_county = {
				order_by = title_province.direction_svalue
				limit = {
					NOT = {
						root = {
							is_target_in_variable_list = {
								name = adjacent_counties
								target = prev
							}
						}
					}
				}
				min = 0
				root = {
					add_to_variable_list = {
						name = adjacent_counties_across_water
						target = prev
					}
				}
			}
		}

		### enemies
		if = {
			limit = {
				scope:player = {
					OR = {
						is_at_war = yes
						AND = {
							exists = liege
							any_liege_or_above = {
								is_at_war = yes
							}
						}
						any_hostile_raider = {
							is_causing_raid_hostility_towards = prev
						}
					}
				}
			}

			scope:player = {
				# LIEGE WARS
				every_liege_or_above = {
					every_war_enemy = {
						limit = {
							NOR = {
								this = scope:player
								is_allied_in_war = scope:player
							}
						}
						every_army = {
							add_army_to_province = yes
							add_to_list = all_enemies
						}
					}
				}

				# MY WARS
				# add all enemies
				every_war_enemy = {
					every_army = {
						add_army_to_province = yes
						add_to_list = all_enemies
					}
					#TODO do we have to?
					# exclude yourself
					every_vassal_or_below = {
						limit = {
							NOR = {
								is_allied_in_war = scope:player
								this = scope:player
							}
						}
						every_army = {
							limit = {
								NOT = { is_in_list = all_enemies }
							}
							add_army_to_province = yes
							add_to_list = all_enemies
						}
					}
				}

				# enemies of my enemies are my enemies
				every_primary_war_enemy = {
					every_war_enemy = {
						limit = {
							NOR = {
								this = scope:player
								is_allied_in_war = scope:player
							}
						}
						every_army = {
							limit = {
								NOT = {
									is_in_list = all_enemies
								}
							}
							add_army_to_province = yes
							add_to_list = all_enemies
						}
					}
				}

				every_hostile_raider = {
					every_army = {
						add_army_to_province = yes
						add_to_list = all_enemies
					}
				}
			}
		}
	}
	is_shown = {
		NOT = {
			has_variable_list = adjacent_counties
		}
	}
}

adjacent_titles = {
	effect = {
		clear_variable_list = adjacent_titles
		title_capital_county.title_province = {
			save_scope_as = origin
		}

		if = {
			limit = { tier = tier_kingdom }
			ordered_title_to_title_neighboring_kingdom = {
				order_by = title_capital_county.title_province.direction_svalue
				min = 0
				root = {
					add_to_variable_list = {
						name = adjacent_titles
						target = prev
					}
				}
			}
		}
		else_if = {
			limit = { tier = tier_empire }
			ordered_title_to_title_neighboring_empire = {
				order_by = title_capital_county.title_province.direction_svalue
				min = 0
				root = {
					add_to_variable_list = {
						name = adjacent_titles
						target = prev
					}
				}
			}
		}
		else_if = {
			limit = { tier = tier_duchy }
			ordered_title_to_title_neighboring_duchy = {
				order_by = title_capital_county.title_province.direction_svalue
				min = 0
				root = {
					add_to_variable_list = {
						name = adjacent_titles
						target = prev
					}
				}
			}
		}
		else_if = {
			limit = { tier = tier_county }
			ordered_title_to_title_neighboring_county = {
				order_by = title_province.direction_svalue
				min = 0
				root = {
					add_to_variable_list = {
						name = adjacent_titles
						target = prev
					}
				}
			}
		}
		else = { }
	}
}
